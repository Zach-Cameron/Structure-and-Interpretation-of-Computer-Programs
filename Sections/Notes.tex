\section{Building Abstractions with Procedures}

\begin{defn}[\idx{computational process}]
    Abstract beings that inhabit computers.
\end{defn}

\begin{defn}[\idx{data}]
    Computational processes manipulate other abstract things called \textbf{data} as they evolve.
\end{defn}

\begin{defn}[\idx{program}]
    A pattern of rules by which the evolution of a process is directed.
\end{defn}

\begin{defn}[\idx{programming languages}]
    That in which programs are carefully composed from symbolic expressions that prescribe the tasks we want our computational processes to perform.
\end{defn}

\begin{defn}[\idx{bugs}, \idx{glitches}]
    Small errors.
\end{defn}
